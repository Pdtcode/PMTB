    ```
    def combine_log_odds(p_xgb: float, p_claude: float, weight_xgb: float = 0.6, weight_claude: float = 0.4) -> float:
        eps = 1e-6
        p_xgb = max(eps, min(1 - eps, p_xgb))
        p_claude = max(eps, min(1 - eps, p_claude))
        logit_xgb = math.log(p_xgb / (1 - p_xgb))
        logit_claude = math.log(p_claude / (1 - p_claude))
        combined = weight_xgb * logit_xgb + weight_claude * logit_claude
        return 1.0 / (1.0 + math.exp(-combined))

    def combine_weighted_average(p_xgb: float, p_claude: float, weight_xgb: float = 0.6, weight_claude: float = 0.4) -> float:
        return max(0.0, min(1.0, weight_xgb * p_xgb + weight_claude * p_claude))
    ```

    ```
    def compute_confidence_interval(
        p_model: float,
        half_width: float = 0.1,
    ) -> tuple[float, float]:
        low = max(0.0, p_model - half_width)
        high = min(1.0, p_model + half_width)
        return (low, high)
    ```

    The CI method is intentionally simple (configurable half-width with boundary clamping). More sophisticated methods (bootstrap, beta distribution) can be swapped in later per Claude's discretion decision. The key invariant is [0, 1] clamping.
